## build

	"C:\Program Files (x86)\Microsoft Visual Studio 14.0\VC\bin\amd64\vcvars64.bat"

    nmake clean
    buildconf --force
    configure --with-all-shared --disable-all --enable-cli ...
    nmake


### memcache

source: https://github.com/php/pecl-caching-memcache

    nmake clean
    buildconf
    configure ^
    --with-all-shared ^
	--disable-all ^
	--enable-cli  ^
        --enable-memcache
    nmake


### ssh2

source: https://github.com/php/pecl-networking-ssh2


    nmake clean
    buildconf
    configure ^
    --with-all-shared ^
	--disable-all ^
	--enable-cli  ^
        --with-ssh2 ^
	--with-openssl
    nmake
